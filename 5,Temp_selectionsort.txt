#include <iostream>
using namespace std;
int n; 
#define size 10

template <class T>

void sel(T A[size]){
int i, j, min;
T temp;
for(i=0;i<n-1;i++){
min=i;
for(j=i+1;j<n;j++){
if(A[j]<A[min]){
min=j;
}
}
temp = A[i];
A[i] = A[min];
A[min] = temp;
}

cout<<"\nSorted Array is: "<<endl;
for(i=0;i<n;i++){
cout<<" "<<A[i];
}

}

int main(){

int A[size];
float B[size];
int i=0;

cout<<"Enter no of integer elements: "<<endl;
cin>>n;
cout<<"Enter integer elements: "<<endl;
for(i=0;i<n;i++){
cin>>A[i];
}
sel(A);
cout<<"\n\n";

cout<<"Enter no of float elements: "<<endl;
cin>>n;
cout<<"\nEnter float elements: "<<endl;
for(i=0;i<n;i++){
cin>>B[i];
}
sel(B);
}
